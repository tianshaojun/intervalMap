let userInfo = {
    name: "小明",
    age: 23,
    sex: '男',
    updateInfo: function () {
      //模拟xmlhttprequest请求延时
      setTimeout(function () {
        this.name = '小王'
        this.age = 30
        this.sex = '女'
      }, 100)
    }
}

userInfo.updateInfo();
